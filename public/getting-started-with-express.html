Getting Started with Express

Express is a minimal and flexible Node.js web application framework that provides a robust set of features for web and mobile applications.

GET Request

To handle GET requests, you define a route using Express's `get()` method.


                const express = require('express');
                const app = express();
        
                app.get('/', (req, res) => {
                    res.send('Hello World!');
                });
        
                app.listen(3000, () => {
                    console.log('Server is running on port 3000');
                });

POST, UPDATE, DELETE, PATCH Requests

For other HTTP methods, you use similar methods like `post()`, `put()`, `delete()`, and `patch()`.

Body Parsing

To parse incoming request bodies, you use middleware like `express.json()` or `express.urlencoded()`.


                const express = require('express');
                const app = express();
        
                app.use(express.json()); // for parsing application/json
                app.use(express.urlencoded({ extended: true })); // for parsing application/x-www-form-urlencoded
        
                app.post('/user', (req, res) => {
                    console.log(req.body);
                    res.json(req.body);
                });
        
                app.listen(3000, () => {
                    console.log('Server is running on port 3000');
                });

Using Middleware

Middleware functions are functions that have access to the request object (`req`), the response object (`res`), and the next middleware function in the applicationâ€™s request-response cycle. Middleware functions can perform tasks, modify `req` and `res` objects, end the request-response cycle, and call the next middleware function in the stack.


                const express = require('express');
                const app = express();
        
                // Custom middleware
                const myMiddleware = (req, res, next) => {
                    console.log('Middleware executed!');
                    next();
                };
        
                app.use(myMiddleware);
        
                app.get('/', (req, res) => {
                    res.send('Hello World!');
                });
        
                app.listen(3000, () => {
                    console.log('Server is running on port 3000');
                });

Enabling JSON and Sending/Receiving JSON

To enable JSON in Express, use `express.json()`. To send and receive JSON, you use `res.json()` and `req.body` respectively.


                const express = require('express');
                const app = express();
        
                app.use(express.json());
        
                app.post('/api/users', (req, res) => {
                    // Create a new user
                    const newUser = req.body;
                    // Assuming some user creation logic here
                    res.json(newUser);
                });
        
                app.get('/api/users/:id', (req, res) => {
                    // Retrieve user by ID
                    const userId = req.params.id;
                    // Assuming fetching user logic here
                    const user = { id: userId, name: 'John Doe' }; // Sample user
                    res.json(user);
                });
        
                app.listen(3000, () => {
                    console.log('Server is running on port 3000');
                });
